## 一、需求概述

### 1.1 背景

根据市场调研，批量生产能力是当前竞品的普遍空白，也是用户的 TOP1 痛点。专业创作者、MCN 机构和电商商家存在高频批量生产需求。

### 1.2 目标用户

| 用户类型 | 核心场景 | 痛点 |
|---------|---------|------|
| 专业创作者 | 批量生成同主题多风格视频 | 逐条生成效率低 |
| MCN 机构 | 达人矩阵号批量分发 | 操作繁琐、无法批量 |
| 电商商家 | 同一商品多平台/多版本视频 | 重复操作浪费时间 |

### 2.2 技术选型

| 组件 | 选型 | 理由 |
|------|------|------|
| **任务队列** | Redis Streams | 已有 Redis，无需额外组件；消费组支持；足够轻量 |
| **进度通知** | WebSocket（SSE）+ 轮询降级 | WebSocket 优先，失败降级 HTTP 轮询 |
| **后台 Worker** | 多进程 Python（`multiprocessing` / `arq`） | FastAPI 原生异步；arq 是轻量级 Redis 队列库 |
| **错误隔离** | 单任务 try/catch 独立包裹 | 单条失败不影响整批 |
| **数据库** | PostgreSQL（已有） | 扩展 `batch_jobs` 和 `batch_tasks` 表 |

### 5.3 Worker 处理流程

```python
# backend/app/workers/script_worker.py

import asyncio
import json
from arq import cron
from arq.connections import RedisSettings
from arq.worker import Worker

async def process_script_generation(ctx, job):
    """处理单个脚本生成任务"""
    task_id = job.args["task_id"]
    batch_job_id = job.args["batch_job_id"]
    input_data = json.loads(job.args["input_data"])

    # 更新任务状态为 running
    await update_task_status(task_id, BatchTaskStatus.RUNNING)

    try:
        # 调用脚本生成服务（已有逻辑）
        generator = ScriptGenerator(session=ctx["session"])
        script = await generator.generate(
            topic=input_data["topic"],
            script_type=ScriptType(input_data["script_type"]),
            quantity=1,
            style=input_data.get("style"),
            user_id=job.args["user_id"],
        )

        # 更新任务结果
        await update_task_result(task_id, result_id=script.id)
        # 通知 WebSocket
        await notify_progress(batch_job_id, task_id, 100, "completed")

    except Exception as e:
        # 单条失败不影响其他任务
        await update_task_error(task_id, str(e))
        await notify_progress(batch_job_id, task_id, 0, "failed")
        # 记录但不抛出异常（防止阻塞队列）

    # 更新主任务统计
    await update_batch_job_stats(batch_job_id)


async def process_video_rendering(ctx, job):
    """处理单个视频渲染任务"""
    task_id = job.args["task_id"]
    batch_job_id = job.args["batch_job_id"]
    input_data = json.loads(job.args["input_data"])

    await update_task_status(task_id, BatchTaskStatus.RUNNING)

    try:
        # 创建视频记录
        video = await create_video_record(input_data)
        await update_task_result(task_id, result_id=video.id)

        # 触发渲染
        await trigger_video_render(video.id)

        # 渲染进度由 VideoRenderer 回调更新
        await notify_progress(batch_job_id, task_id, 50, "running")

    except Exception as e:
        await update_task_error(task_id, str(e))
        await notify_progress(batch_job_id, task_id, 0, "failed")


# Worker 配置
class WorkerSettings:
    redis_settings = RedisSettings.from_url("redis://localhost:6379/0")
    functions = [process_script_generation, process_video_rendering]
    max_concurrent_tasks = 5
    keep_result = 60 * 60  # 保留结果 1 小时
```

### 5.4 进度回调机制

```python
# backend/app/services/progress_notifier.py

from typing import Optional
from fastapi import WebSocket

class ProgressNotifier:
    """进度通知器（WebSocket + 轮询降级）"""

    def __init__(self):
        # 内存中的 WebSocket 连接池（按 batch_job_id 索引）
        self._connections: dict[str, list[WebSocket]] = {}

    async def connect(self, batch_job_id: str, ws: WebSocket):
        """客户端连接 WebSocket"""
        await ws.accept()
        self._connections.setdefault(batch_job_id, []).append(ws)

    def disconnect(self, batch_job_id: str, ws: WebSocket):
        """客户端断开连接"""
        if batch_job_id in self._connections:
            self._connections[batch_job_id].remove(ws)

    async def broadcast(self, batch_job_id: str, message: dict):
        """广播消息给所有连接的客户端"""
        if batch_job_id not in self._connections:
            return

        disconnected = []
        for ws in self._connections[batch_job_id]:
            try:
                await ws.send_json(message)
            except Exception:
                disconnected.append(ws)

        # 清理断开的连接
        for ws in disconnected:
            self.disconnect(batch_job_id, ws)

    async def notify_progress(self, batch_job_id: str, task_id: str,
                              progress: int, status: str,
                              result_id: Optional[str] = None,
                              error: Optional[str] = None):
        """通知进度变化"""
        message = {
            "type": "progress" if status == "running" else status,
            "task_id": task_id,
            "progress": progress,
            "status": status,
        }
        if result_id:
            message["result_id"] = result_id
        if error:
            message["error"] = error

        await self.broadcast(batch_job_id, message)

# 全局单例
notifier = ProgressNotifier()
```

### 9.1 进度 Hook

```typescript
// frontend/src/hooks/useBatchProgress.ts

import { useEffect, useState, useRef } from 'react'
import { batchApi } from '@/services/api'

interface TaskProgress {
  id: string
  status: 'pending' | 'running' | 'completed' | 'failed' | 'skipped'
  progress: number
  result_id?: string
  error_message?: string
}

interface BatchProgress {
  batch_job_id: string
  status: 'pending' | 'running' | 'completed' | 'failed' | 'partially_failed'
  total_count: number
  success_count: number
  failed_count: number
  tasks: TaskProgress[]
}

export function useBatchProgress(batchJobId: string, enabled: boolean = true) {
  const [progress, setProgress] = useState<BatchProgress | null>(null)
  const wsRef = useRef<WebSocket | null>(null)
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null)

  // WebSocket 连接
  useEffect(() => {
    if (!enabled || !batchJobId) return

    const wsUrl = `${window.location.protocol === 'https:' ? 'wss:' : 'ws:'}//${window.location.host}/api/v1/ws/batch/${batchJobId}`
    const ws = new WebSocket(wsUrl)

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data)
      // 更新本地状态（实时进度由 WebSocket 推送）
      setProgress(prev => {
        if (!prev) return prev
        const taskIndex = prev.tasks.findIndex(t => t.id === data.task_id)
        if (taskIndex >= 0) {
          const updatedTasks = [...prev.tasks]
          updatedTasks[taskIndex] = { ...updatedTasks[taskIndex], ...data }
          return { ...prev, tasks: updatedTasks }
        }
        return prev
      })
    }

    ws.onclose = () => {
      // WebSocket 断开时降级到轮询
      startPolling()
    }

    wsRef.current = ws

    return () => {
      ws.close()
      if (pollRef.current) clearInterval(pollRef.current)
    }
  }, [batchJobId, enabled])

  // 轮询降级
  const startPolling = () => {
    pollRef.current = setInterval(async () => {
      try {
        const response = await batchApi.getBatchJobStatus(batchJobId)
        setProgress(response.data)
        if (response.data.status === 'completed' || response.data.status === 'failed') {
          if (pollRef.current) clearInterval(pollRef.current)
        }
      } catch (e) {
        console.error('Poll error:', e)
      }
    }, 3000)
  }

  // 初始加载
  useEffect(() => {
    if (!enabled || !batchJobId) return
    batchApi.getBatchJobStatus(batchJobId).then(res => setProgress(res.data))
  }, [batchJobId, enabled])

  return { progress }
}
```

```python
# backend/app/services/batch_service.py

import json
from typing import List
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.batch import BatchJob, BatchTask, BatchJobType, BatchJobStatus, BatchTaskType, BatchTaskStatus
from app.models.script import ScriptType
from app.core.id_generator import generate_id
from app.core.database import async_session
from app.services.progress_notifier import notifier


class BatchService:
    """批量任务服务"""

    def __init__(self, session: AsyncSession):
        self.session = session

    async def create_script_generation_job(
        self,
        user_id: str,
        items: List[dict],
    ) -> BatchJob:
        """创建批量脚本生成任务"""
        job_id = generate_id("batch_jobs")

        # 创建主任务记录
        job = BatchJob(
            id=job_id,
            user_id=user_id,
            job_type=BatchJobType.SCRIPT_GENERATION,
            status=BatchJobStatus.PENDING,
            total_count=len(items),
            config_json={"items": items},
        )
        self.session.add(job)

        # 创建子任务记录
        for item in items:
            task = BatchTask(
                id=generate_id("batch_tasks"),
                batch_job_id=job_id,
                task_type=BatchTaskType.SCRIPT,
                status=BatchTaskStatus.PENDING,
                input_data=item,
            )
            self.session.add(task)

        await self.session.commit()
        await self.session.refresh(job)

        # 异步分发到 Redis 队列
        await self._dispatch_script_tasks(job_id, items, user_id)

        return job

    async def create_video_generation_job(
        self,
        user_id: str,
        script_ids: List[str],
        template_id: str | None,
        material_ids: List[str],
    ) -> BatchJob:
        """创建批量视频生成任务"""
        job_id = generate_id("batch_jobs")

        items = [
            {"script_id": sid, "template_id": template_id, "material_ids": material_ids}
            for sid in script_ids
        ]

        job = BatchJob(
            id=job_id,
            user_id=user_id,
            job_type=BatchJobType.VIDEO_RENDERING,
            status=BatchJobStatus.PENDING,
            total_count=len(script_ids),
            config_json={"script_ids": script_ids, "template_id": template_id, "material_ids": material_ids},
        )
        self.session.add(job)

        for item in items:
            task = BatchTask(
                id=generate_id("batch_tasks"),
                batch_job_id=job_id,
                task_type=BatchTaskType.VIDEO,
                status=BatchTaskStatus.PENDING,
                input_data=item,
            )
            self.session.add(task)

        await self.session.commit()
        await self.session.refresh(job)

        await self._dispatch_video_tasks(job_id, items, user_id)

        return job

    async def _dispatch_script_tasks(self, job_id: str, items: List[dict], user_id: str):
        """分发脚本生成任务到 Redis 队列"""
        import redis.asyncio as redis
        from app.core.config import settings

        r = redis.from_url(settings.REDIS_URL)
        queue_name = "batch:queue:script_generation"

        for i, item in enumerate(items):
            task = BatchTask(
                id=generate_id("batch_tasks"),
                batch_job_id=job_id,
                task_type=BatchTaskType.SCRIPT,
                status=BatchTaskStatus.PENDING,
                input_data=item,
                queue_name=queue_name,
            )
            self.session.add(task)

            msg_id = await r.xadd(
                queue_name,
                {
                    "task_id": task.id,
                    "batch_job_id": job_id,
                    "task_type": "script",
                    "user_id": user_id,
                    "input_data": json.dumps(item),
                    "retry_count": "0",
                }
            )
            task.queue_id = msg_id

        await self.session.commit()

        # 更新主任务状态
        await self._update_job_status(job_id, BatchJobStatus.RUNNING)

        # 启动 Worker（如果尚未运行）
        # 生产环境通过 systemd/supervisor 管理 Worker 进程
        await r.close()

    async def _dispatch_video_tasks(self, job_id: str, items: List[dict], user_id: str):
        """分发视频渲染任务到 Redis 队列"""
        import redis.asyncio as redis
        from app.core.config import settings

        r = redis.from_url(settings.REDIS_URL)
        queue_name = "batch:queue:video_rendering"

        for item in items:
            task = BatchTask(
                id=generate_id("batch_tasks"),
                batch_job_id=job_id,
                task_type=BatchTaskType.VIDEO,
                status=BatchTaskStatus.PENDING,
                input_data=item,
                queue_name=queue_name,
            )
            self.session.add(task)

            msg_id = await r.xadd(
                queue_name,
                {
                    "task_id": task.id,
                    "batch_job_id": job_id,
                    "task_type": "video",
                    "user_id": user_id,
                    "input_data": json.dumps(item),
                    "retry_count": "0",
                }
            )
            task.queue_id = msg_id

        await self.session.commit()
        await self._update_job_status(job_id, BatchJobStatus.RUNNING)
        await r.close()

    async def _update_job_status(self, job_id: str, status: BatchJobStatus):
        """更新主任务状态"""
        job = await self.session.get(BatchJob, job_id)
        if job:
            job.status = status
            job.updated_at = datetime.utcnow()
            if status in (BatchJobStatus.COMPLETED, BatchJobStatus.FAILED, BatchJobStatus.PARTIALLY_FAILED):
                job.completed_at = datetime.utcnow()
            await self.session.commit()

    async def get_job_status(self, job_id: str, user_id: str) -> BatchJob | None:
        """获取批量任务状态（含子任务详情）"""
        job = await self.session.get(BatchJob, job_id)
        if not job or job.user_id != user_id:
            return None

        tasks = await self.session.execute(
            select(BatchTask).where(BatchTask.batch_job_id == job_id).order_by(BatchTask.created_at)
        )
        job.tasks = tasks.scalars().all()
        return job
```

## 十一、错误处理策略

### 11.1 单条失败隔离

```python
# Worker 中的错误处理
async def process_script_generation(ctx, job):
    task_id = job.args["task_id"]

    try:
        # 单条任务的完整逻辑包裹在 try 中
        result = await _do_generate_script(job.args)
        await update_task_result(task_id, result_id=result.id)

    except Exception as e:
        # 单条失败 → 记录错误，继续处理下一条
        await update_task_error(task_id, str(e))
        await notify_progress(job.args["batch_job_id"], task_id, 0, "failed")
        # 注意：这里没有 raise，整个任务不会中断

    finally:
        # 更新统计
        await update_batch_job_stats(job.args["batch_job_id"])
```

```python
# 配置重试策略
async def process_script_generation(ctx, job):

    if retry_count >= 3:
```

### 11.3 主任务状态判定

```python
async def update_batch_job_stats(job_id: str):
    """更新主任务统计和状态"""
    async with async_session() as session:
        result = await session.execute(
            select(
                func.count(BatchTask.id).label("total"),
                func.sum(sql_case((BatchTask.status == BatchTaskStatus.COMPLETED, 1), else_=0)).label("success"),
                func.sum(sql_case((BatchTask.status == BatchTaskStatus.FAILED, 1), else_=0)).label("failed"),
            ).where(BatchTask.batch_job_id == job_id)
        )
        row = result.one()

        job = await session.get(BatchJob, job_id)
        job.total_count = row.total
        job.success_count = row.success
        job.failed_count = row.failed

        # 判定主任务状态
        if row.total == row.success:
            job.status = BatchJobStatus.COMPLETED
        elif row.total == row.failed:
            job.status = BatchJobStatus.FAILED
        elif row.success > 0:
            job.status = BatchJobStatus.PARTIALLY_FAILED
        else:
            job.status = BatchJobStatus.RUNNING

        await session.commit()
```
